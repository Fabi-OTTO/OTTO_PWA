#include <SoftwareSerial.h>
SoftwareSerial SerialBt(2, 3);
void setup()
{
Serial.begin(9600);
SerialBt.begin(9600);
}
void loop()
{
if (SerialBt.available()) {
Serial.write(SerialBt.read());
}
if (Serial.available()) {
SerialBt.write(Serial.read());
}
}
